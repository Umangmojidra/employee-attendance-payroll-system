CREATE TABLE employees (
  e_id INT PRIMARY KEY AUTO_INCREMENT,
  e_name VARCHAR(50),
  dep_id int,
  mobile BIGINT,
  email VARCHAR(50),
  city VARCHAR(40),
  joining_date DATE DEFAULT (CURRENT_DATE),
  blood_group VARCHAR(10),
  salary DECIMAL(10,2)
)auto_increment = 202501001;

create table departments (dep_id int primary key, dep_name varchar(40));

create table attendance(e_id INT,login TIMESTAMP,status VARCHAR(40), penalty BOOLEAN DEFAULT 0,overtime INT DEFAULT 0, FOREIGN KEY(e_id) REFERENCES employees (e_id),FOREIGN KEY(dep_id) REFERENCES departments (dep_id));

create table manager(m_id INT PRIMARY KEY not null, m_name VARCHAR(40),pin INT not null);
